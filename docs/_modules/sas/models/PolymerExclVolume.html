    def _polymerexclvol(self, x):
        
        sc = self.params['scale']
        rg = self.params['rg']
        mm  = self.params['m']
        bg = self.params['background']
    
        nu = 1.0 / mm
    
        Xx = x * x * rg * rg *(2.0 * nu + 1.0) * (2.0 * nu + 2.0) / 6.0
        onu = 1.0 / nu
        o2nu = 1.0 /(2.0 * nu)
        Ps =(1.0 / (nu * pow(Xx,o2nu))) * (gamma(o2nu)*gammainc(o2nu,Xx) - \
                        1.0 / pow(Xx,o2nu) * gamma(onu)*gammainc(onu,Xx))

        if x == 0:
            Ps = 1.0

        return (sc * Ps + bg);
